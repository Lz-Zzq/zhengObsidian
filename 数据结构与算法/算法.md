### 二分查找
```
//寻找最左侧  
while (i < j) {  
    int m = (i + j) >>> 1;  
	if (target <= a[m]) { //直接将=移动到此处 查找最左值  
		candidate = j = m;  
	} else {  
		i = m + 1;  
	}  
} 
return candidate;
//寻找最右边侧  
while (i < j) {  
	int m = (i + j) >>> 1;  
		if (target < a[m]) { //直接将=移动到此处 查找最左值  
			j = m;  
	} else {  
		candidate = i = m + 1;  
	}  
}  
return candidate;
//正常二分查找  
while (i < j) {  
	int m = (i + j) >>> 1;  
	if (target < a[m]) {  
		j = m;  
	} else if (a[m] < target) {  
		i = m + 1;  
	} else {  
		return candidate = m;  
		// i = m+1; //最右边  
		// j = m; //最左边  
	}  
}

//正常版本 j参与运算
int i = 0, j = a.length - 1;  
while (i <= j) { //只要i<j 就可以继续执行 i = j 也可以执行最后一次查看是否正确  
	int m = (i + j) >>> 1;  
	if (target < a[m]) {  
		j = m - 1;  
	} else if (a[m] < target) {  
		i = m + 1;  
	} else {  
		return m;  
	}  
}  
return -1;
```

### 二分查找
```
//寻找最左侧  
while (i < j) {  
    int m = (i + j) >>> 1;  
	if (target <= a[m]) { //直接将=移动到此处 查找最左值  
		candidate = j = m;  
	} else {  
		i = m + 1;  
	}  
}  
//寻找最右边侧  
while (i < j) {  
	int m = (i + j) >>> 1;  
		if (target < a[m]) { //直接将=移动到此处 查找最左值  
			j = m;  
	} else {  
		candidate = i = m + 1;  
	}  
}  
  
//正常二分查找  
while (i < j) {  
	int m = (i + j) >>> 1;  
	if (target < a[m]) {  
		j = m;  
	} else if (a[m] < target) {  
		i = m + 1;  
	} else {  
//将m指向的目录记录  
	return candidate = m;  
	// i = m+1; //最右边  
	// j = m; //最左边  
	}  
}  
	 return candidate;  
```

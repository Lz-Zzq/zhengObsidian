  //左侧查找
    public int leftMethod(int[] nums,int target){
        int left = 0,right = nums.length - 1;
        int candidate = -1;
        while(right >= left ){
            int m = (left+right) >>> 1;
            if(target < nums[m]){
                right = m - 1;
            }else if(target > nums[m]){
                left = m + 1;
            }else{
                candidate = m;
                right = m-1;
            }
        }
        return candidate;
    }
//右侧查找
    public int rightMethod(int[] nums,int target){
        int left = 0,right = nums.length-1;
        int candidate = -1;
        while(right >= left ){
            int m = (left+right) >>> 1;
            if(target < nums[m]){
                right = m - 1;
            }else if(target > nums[m]){
                left = m + 1;
            }else{
               candidate = m;
               left = m + 1;
            }
        }
        return candidate;
    }
